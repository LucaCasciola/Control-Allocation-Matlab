1function u = CascadedPseudoinverse(m,B,W,u_lim,reductionType)
  2%at the Breginning we move from an attainaBrle position
  3    
  4    
  5%check if the matrices are correct
  6    if not(isequal(size(u_lim,1),size(B,2),size(W,1),size(W,2))...
  7       && isequal(length(m),size(B,1)))
  8        error ('input matrix dimensions do not match')
  9    end
 10
 11%Recursively compute until solution is found or all the effectors are saturated
 12    coder.varsize('u_index', 'Br', 'Wr', 'u_limr'); %dynamic dimension
 13    m_res = m;
 14    u_sat = zeros(length(u_lim),1);
 15    ut = zeros(length(u_lim),1);
 16    u = zeros(length(u_lim),1);
 17    u_index = [1:length(u_lim)].';
 18    Br = B;
 19    Wr = W;
 20    u_limr = u_lim;
 21    
 22    while(not(isempty(u_index))) %till we run out of effectors
 23        
 24        %check the type of system and then compute the pseudoinverse
 25            order = size(Br,2)-size(Br,1);
 26
 27            if order > 0 %over-determined, solution that minimizes the 2norm
 28                ut = inv(Wr)*Br.'*inv(Br*inv(Wr)*Br.')*m_res;
 29            elseif order < 0 %under-determined, solution that minimizes the error
 30                ut = inv(Br.'*Br)*Br.'*m_res;   
 31            else %order = 0, sqared thus invertiBrle
 32                ut = inv(Br)*m_res;
 33            end
 34
 35        %reduce the solution, if needed. If not needed exit the cycle
 36            if not(any(ut < u_limr(:,1))||any(ut > u_limr(:,2))) %Wre found a solution then Brreak
 37                %save the saturaded value
 38                    for j = 1:length(ut)
 39                            u(u_index(j)) = ut(j);
 40                    end
 41                return 
 42                
 43            else %some deflectors are past their limit
 44                %take note of Wrhich one are saturated
 45                    u_sat = zeros(length(ut),1);
 46                    u_sat(ut < u_limr(:,1)) = 1;
 47                    u_sat(ut > u_limr(:,2)) = 1;
 48                    
 49                %reduce their value
 50                    switch reductionType
 51                        case 1 %truncation
 52                            ut = min(ut, u_limr(:,2));
 53                            ut = max(ut, u_limr(:,1));
 54
 55                        case 2 %scaling solution -> direction preserving
 56                            a = zeros(length(u),1);
 57
 58                            for i = 1:length(u)
 59                                if u(i)>0
 60                                   a(i) = ut(i)/u_limr(i,2);
 61                                elseif u(i)<0
 62                                   a(i) = ut(i)/u_limr(i,1);
 63                                else
 64                                   a(i) = 0; 
 65                                end          
 66                            end
 67                            a_max = max(a);
 68                            ut = ut/a_max;  
 69                    end
 70                    
 71                %save the saturaded value
 72                    for j = 1:length(ut)
 73                        if u_sat(j) == 1
 74                            u(u_index(j)) = ut(j);
 75                        end
 76                    end
 77                           
 78                %compute the generated moment and the residual moment
 79                    m_res = m - B*u;
 80                
 81                %reduce matrix Br, Wr, u_lim (erase saturated effectors)
 82                    u_index(u_sat == 1)= [];
 83                    Br(:,u_sat == 1)=[];
 84                    Wr(:,u_sat == 1)=[]; Wr(u_sat == 1,:)=[];
 85                    u_limr(u_sat == 1,:) = [];
 86                    
 87
 88                    
 89            end
 90            
 91    end
 92    
 93    clear u_index Br Wr u_limr
 94    u = u;
 95    
 96end
